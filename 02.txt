def string_middle(str):
    buf = len(str)
    value = int(buf / 2)
    test = str[value]
    test1 = str[value - 1 : value + 1]
    if buf % 2 == 1:
        return test
    elif buf % 2 == 0:
        return test1

print(string_middle("power"))
